193 │ $ Algorithm: Particle Physics 194 │ 
Particle Motion Algorithm:
 195 │   1. Initialize N particles with random positions and velocities
 196 │   2. Each frame:
 197 │      a. Calculate distance from particle to mouse cursor
 198 │      b. If distance < attractionRadius: apply attraction force
 199 │      c. If distance < repulsionRadius: apply repulsion force
 200 │      d. Update velocity using F = ma (Newton's Second Law)
 201 │      e. Apply drag/friction to velocity
 202 │      f. Update position: position += velocity
 203 │      g. Wrap particles around screen edges
 204 │      h. Draw particle with age-based opacity
 205 │ 
 206 │ Key Physics:
 207 │   - Force magnitude decreases with distance squared (inverse square law)
 208 │   - Drag coefficient dampens oscillations (natural friction)
 209 │   - Color changes based on particle age (lifecycle visualization)
 210 │   - Trail effect creates motion blur perception

// PARTICLE SYSTEM with p5.js
 217 │ // Demonstrates emergent behavior from simple physics rules
 218 │ 
 219 │ let particles = [];
 220 │ const PARTICLE_COUNT = 100;
 221 │ const ATTRACTION_RADIUS = 150;
 222 │ const REPULSION_RADIUS = 50;
 223 │ const MAX_FORCE = 0.5;
 224 │ const DRAG = 0.95;
 225 │ 
 226 │ function setup() {
 227 │   let container = document.getElementById('sketch-container');
 228 │   let w = container.offsetWidth;
 229 │   let h = 500;
 230 │ 
 231 │   let sketchCanvas = createCanvas(w, h);
 232 │   sketchCanvas.parent('sketch-container');
 233 │ 
 234 │   // Initialize particles
 235 │   for (let i = 0; i < PARTICLE_COUNT; i++) {
 236 │     particles.push(new Particle(
 237 │       random(w),
 238 │       random(h)
 239 │     ));
 240 │   }
 241 │ }
 242 │ 
 243 │ function draw() {
 244 │   background(0, 0, 0, 25);
 245 │ 
 246 │   // Update and draw particles
 247 │   for (let particle of particles) {
 248 │     particle.update(mouseX, mouseY);
 249 │     particle.display();
 250 │   }
 251 │ 
 252 │   // Draw cursor indicator
 253 │   stroke(191, 255, 0, 100);
 254 │   strokeWeight(2);
 255 │   noFill();
 256 │   circle(mouseX, mouseY, ATTRACTION_RADIUS * 2);
 257 │   circle(mouseX, mouseY, REPULSION_RADIUS * 2);
 258 │ }
 259 │ 
 260 │ class Particle {
 261 │   constructor(x, y) {
 262 │     this.pos = createVector(x, y);
 263 │     this.vel = p5.Vector.random2D().mult(2);
 264 │     this.acc = createVector(0, 0);
 265 │     this.age = 0;
 266 │   }
 267 │ 
 268 │   update(targetX, targetY) {
 269 │     let target = createVector(targetX, targetY);
 270 │     let distance = dist(
 271 │       this.pos.x, this.pos.y,
 272 │       targetX, targetY
 273 │     );
 274 │ 
 275 │     // Attraction force
 276 │     if (distance < ATTRACTION_RADIUS && distance > REPULSION_RADIUS) {
 277 │       let direction = p5.Vector.sub(target, this.pos);
 278 │       direction.normalize();
 279 │       let force = direction.mult(MAX_FORCE * 0.3);
 280 │       this.acc.add(force);
 281 │     }
 282 │ 
 283 │     // Repulsion force
 284 │     if (distance < REPULSION_RADIUS) {
 285 │       let direction = p5.Vector.sub(this.pos, target);
 286 │       direction.normalize();
 287 │       let force = direction.mult(MAX_FORCE);
 288 │       this.acc.add(force);
 289 │     }
 290 │ 
 291 │     // Apply physics
 292 │     this.vel.add(this.acc);
 293 │     this.vel.mult(DRAG);
 294 │     this.pos.add(this.vel);
 295 │     this.acc.mult(0);
 296 │ 
 297 │     // Wrap around edges
 298 │     this.pos.x = (this.pos.x + width) % width;
 299 │     this.pos.y = (this.pos.y + height) % height;
 300 │ 
 301 │     this.age++;
 302 │   }
 303 │ 
 304 │   display() {
 305 │     let opacity = map(this.age, 0, 300, 255, 0);
 306 │     let speed = this.vel.mag();
 307 │     let hueValue = map(speed, 0, 5, 120, 50);
 308 │ 
 309 │     colorMode(HSB);
 310 │     stroke(hueValue, 100, 100, opacity);
 311 │     strokeWeight(2);
 312 │     point(this.pos.x, this.pos.y);
 313 │     colorMode(RGB);
 314 │   }
 315 │ }

320 │ $ Tunable Parameters 321 │ 
PARTICLE_COUNT = 100
 322 │   Increase for more visible swarms, decrease for cleaner animation
 323 │ 
 324 │ ATTRACTION_RADIUS = 150
 325 │   Distance at which particles begin accelerating toward cursor
 326 │   Larger values = particles respond from further away
 327 │ 
 328 │ REPULSION_RADIUS = 50
 329 │   Inner radius where particles flee from cursor
 330 │   Creates the "avoid" zone around cursor
 331 │ 
 332 │ MAX_FORCE = 0.5
 333 │   Maximum acceleration magnitude per frame
 334 │   Higher values = snappier, more responsive movement
 335 │ 
 336 │ DRAG = 0.95
 337 │   Velocity dampening factor (0-1)
 338 │   Lower = more friction, smoother motion
 339 │   Higher = less friction, more momentum

Emergent Behavior
345 │
346 │ This system demonstrates how complex, organic-looking behavior emerges from simple rules. 347 │ No particle "knows" about others—they only respond to the mouse cursor and physics. 348 │ The collective swarm behavior appears intelligent yet requires no explicit coordination. 349 │ This is the basis of flocking algorithms used in games, animation, and robotics. 350 │
