194 │ $ Algorithm: Perlin Noise Flow Field 195 │ 
Noise Field Algorithm:
 196 │   1. Create a 2D grid of flow vectors
 197 │   2. Initialize particles at random positions
 198 │   3. Each frame:
 199 │      a. Sample Perlin noise at each grid point: noise3D(x, y, time)
 200 │      b. Convert noise value to angle: angle = noise * TWO_PI
 201 │      c. Create vector from angle: direction = [cos(angle), sin(angle)]
 202 │      d. For each particle:
 203 │         - Look up flow direction at particle position
 204 │         - Accelerate particle in that direction
 205 │         - Update position and velocity
 206 │         - Draw particle as line segment (direction indicator)
 207 │         - Wrap around screen edges
 208 │ 
 209 │ Key Concepts:
 210 │   - Perlin Noise: Smooth, continuous pseudo-random values
 211 │   - Time dimension: noise evolves over time for continuous variation
 212 │   - Scale parameter: Controls "zoom" of noise (larger = more variation)
 213 │   - Flow visualization: Arrows/lines show direction and magnitude
 214 │ 
 215 │ Properties:
 216 │   - Deterministic: Same seed produces same pattern
 217 │   - Continuous: Values change smoothly, no abrupt transitions
 218 │   - Multi-scale: Can layer multiple noise octaves (Fractional Brownian Motion)

// PERLIN NOISE FLOW FIELD with p5.js
 225 │ // Particles follow the direction of Perlin noise gradients
 226 │ 
 227 │ let particles = [];
 228 │ const PARTICLE_COUNT = 200;
 229 │ const NOISE_SCALE = 0.015;
 230 │ const PARTICLE_SPEED = 2;
 231 │ const TRAIL_LENGTH = 3;
 232 │ 
 233 │ let timeOffset = 0;
 234 │ 
 235 │ function setup() {
 236 │   let container = document.getElementById('sketch-container');
 237 │   let w = container.offsetWidth;
 238 │   let h = 500;
 239 │ 
 240 │   let sketchCanvas = createCanvas(w, h);
 241 │   sketchCanvas.parent('sketch-container');
 242 │ 
 243 │   // Initialize particles
 244 │   for (let i = 0; i < PARTICLE_COUNT; i++) {
 245 │     particles.push(new FlowParticle(
 246 │       random(w),
 247 │       random(h)
 248 │     ));
 249 │   }
 250 │ }
 251 │ 
 252 │ function draw() {
 253 │   background(0, 0, 0, 15);
 254 │ 
 255 │   for (let particle of particles) {
 256 │     let angle = noise(
 257 │       particle.pos.x * NOISE_SCALE,
 258 │       particle.pos.y * NOISE_SCALE,
 259 │       timeOffset
 260 │     ) * TWO_PI;
 261 │ 
 262 │     particle.flowDirection = p5.Vector.fromAngle(angle);
 263 │     particle.update();
 264 │     particle.display();
 265 │   }
 266 │ 
 267 │   timeOffset += 0.002;
 268 │ }
 269 │ 
 270 │ class FlowParticle {
 271 │   constructor(x, y) {
 272 │     this.pos = createVector(x, y);
 273 │     this.vel = createVector(0, 0);
 274 │     this.flowDirection = createVector(1, 0);
 275 │     this.trailLength = TRAIL_LENGTH;
 276 │   }
 277 │ 
 278 │   update() {
 279 │     // Accelerate toward flow direction
 280 │     this.vel.lerp(this.flowDirection, 0.1);
 281 │     this.vel.limit(PARTICLE_SPEED);
 282 │ 
 283 │     // Update position
 284 │     this.pos.add(this.vel);
 285 │ 
 286 │     // Wrap around edges
 287 │     this.pos.x = (this.pos.x + width) % width;
 288 │     this.pos.y = (this.pos.y + height) % height;
 289 │   }
 290 │ 
 291 │   display() {
 292 │     // Draw particle as direction indicator line
 293 │     let speed = this.vel.mag();
 294 │     let opacity = map(speed, 0, PARTICLE_SPEED, 50, 200);
 295 │ 
 296 │     stroke(0, 255, 0, opacity);
 297 │     strokeWeight(1.5);
 298 │     line(
 299 │       this.pos.x,
 300 │       this.pos.y,
 301 │       this.pos.x + this.flowDirection.x * this.trailLength,
 302 │       this.pos.y + this.flowDirection.y * this.trailLength
 303 │     );
 304 │ 
 305 │     // Optional: draw particle dot
 306 │     point(this.pos.x, this.pos.y);
 307 │   }
 308 │ }

313 │ $ Tunable Parameters 314 │ 
PARTICLE_COUNT = 200
 315 │   Number of particles in the field
 316 │   More particles = denser visualization, slower performance
 317 │ 
 318 │ NOISE_SCALE = 0.015
 319 │   Controls the "zoom level" of the noise function
 320 │   Smaller values = larger noise features (more swirling)
 321 │   Larger values = finer detail (more chaotic)
 322 │   Typical range: 0.005 - 0.05
 323 │ 
 324 │ PARTICLE_SPEED = 2
 325 │   Maximum velocity magnitude for particles
 326 │   Higher = faster, more energetic movement
 327 │ 
 328 │ TRAIL_LENGTH = 3
 329 │   Length of the direction indicator line
 330 │   Longer lines make direction more visible
 331 │ 
 332 │ timeOffset increment (currently 0.002)
 333 │   How fast the noise field evolves
 334 │   Lower = slower evolution, more time-stretched patterns
 335 │   Higher = faster evolution, more chaotic change

Perlin Noise Fundamentals
341 │
342 │ Perlin noise was invented by Ken Perlin in 1983 and is fundamental to procedural generation. 343 │ Unlike random noise, Perlin noise produces smooth, continuous values that change gradually. 344 │ The 3D sampling (x, y, time) creates an animated flow field that evolves naturally over time. 345 │ This technique is used in game terrain generation, cloud simulation, and digital art. 346 │

// Multi-octave noise for richer detail
 353 │ // Each octave adds finer detail with reduced amplitude
 354 │ 
 355 │ function fbm(x, y, time) {
 356 │   let value = 0;
 357 │   let amplitude = 1;
 358 │   let frequency = 1;
 359 │   let maxValue = 0;
 360 │ 
 361 │   for (let i = 0; i < 4; i++) {
 362 │     value += noise(
 363 │       x * frequency,
 364 │       y * frequency,
 365 │       time
 366 │     ) * amplitude;
 367 │ 
 368 │     maxValue += amplitude;
 369 │     amplitude *= 0.5;
 370 │     frequency *= 2;
 371 │   }
 372 │ 
 373 │   return value / maxValue;
 374 │ }
